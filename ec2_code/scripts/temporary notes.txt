Will need to configure RCON for clean communication with the factorio server:
eg.
const Rcon = require('rcon-client');

async function getPlayerCount() {
    const client = new Rcon.Client({
        host: 'localhost',          // Address of the Factorio server
        port: 27015,                // The RCON port
        password: 'your_rcon_password' // The RCON password set in config.ini
    });

    try {
        await client.connect();
        console.log('Connected to Factorio server.');

        // Send the command to get the player count
        const response = await client.send('/players');
        
        // Log the response to see the player count (may need parsing)
        console.log('Server response:', response);
        client.end();
        
        // Parse the response for the player count (may need adjustments)
        const playerCount = (response.match(/players: (\d+)/) || [])[1];
        console.log(`Player count: ${playerCount}`);
    } catch (error) {
        console.error('Error connecting to the Factorio server:', error);
    }
}

getPlayerCount();
